\section{Clitic:REAS \emph{\textipa{VOOka}}}
	\begin{example}
	\label{ex:reas}
		Na'arimo faavugoo kariwooka [\textipa{naPaRi FaaBugOO kaRiVOOka}]
		\gll na- \textipa{P}a\R i {m\OO} {\F aa\B u} {-g\OO\OO} {ka\R i} -\W\OO\OO ka
		1- go {-\pst} home {-\dirprox} sleep -\reas
		\glt `I went home in order to sleep.'
		\glend
	\end{example}

\section{V:desire \emph{\textipa{gOPi}}}
	\begin{example}
	\label{ex:desire}
		Nago'i o'ati. [\textipa{nagoPi PoPati}]
		\gll na- go\textipa{P}i \textipa{PoPati}
		1- desire dog
		\glt `I want a dog.'
		\glend
	\end{example}

	\begin{example}
	\label{ex:desire_imp}
		Nigo'i o'ati ri? [\textipa{nigoPi PoPati Ri}]
		\gll ni- go\textipa{P}i \textipa{PoPati} \R i
		2- desire dog {\interr}
		\glt `Do you want a dog?'
		\glend
	\end{example}

\section{V:be.able.to \emph{\textipa{saRu}}}
	\begin{example}
	\label{ex:be.able.to}
		Nasarunga ari favukoo, nimiromi siwooka wakoo. [\textipa{nasaRuNa PaRi FaBukOO nimiROmi siVOOka VakO}]
		\gll na- saru -\N a \textipa{PaRi} {\F a\B u} {-k\OO\OO} ni- {mi\R\OO} -mi si -\W\OO\OO ka \W a -k\OO\OO
		1- be.able.to {-\negative} go house {-\dirprox} 1- cat {-\poss} {\cop} {-\reas} that -\dirprox
		\glt `I cannot go to your house, as your cat is there.'
		\glend
	\end{example}
